7- write a function that accept parameter n and return factorial of n 

    var n=prompt("enter number"){
        let answer = n;
        if (n == 0 || n == 1){
          return answer;
        }else{
          for(var i = n; i >= 1; i--){
            answer = answer * i;
          }
          return answer;
        }  
      }
      console.log(n);
